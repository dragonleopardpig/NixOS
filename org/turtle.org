           "orange",

           "green",
           "blue",
           "indigo",
           "violet",
           "cyan",
           "magenta",
           "black",
           ]

#+BEGIN_SRC python  
def concentricCircle1(start, end, step):
    counter = 0
    while counter < len(rainbow):
        turtle.color(str(rainbow[counter]))
        turtle.circle(start + counter * step)
        counter = counter + 1
        turtle.up()
        turtle.right(90)
        turtle.forward(step)
        turtle.down()
        turtle.left(90)
