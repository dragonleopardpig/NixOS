** Roots of an Equation
#+BEGIN_SRC jupyter-python  
from scipy.optimize import root
from math import cos

def eqn(x):
    x = x.item() #convert x from array back to scalar value
    return x + cos(x)

myroot = root(eqn, 0)

print(myroot.x)
#+END_SRC

#+RESULTS:
: [-0.73908513]
